16  package com.mycompany.flokeszownengine.BL;
17  
18  import javafx.scene.image.Image;
19  import javafx.scene.image.ImageView;
20  import static java.lang.Math.abs;
21  import java.util.Random;
22  /**
23   *
24   * @author kiss
25   */
26  public class Bear {
27      Image image1;
28      ImageView bear;
29      int knocked;
30      int bearPoz[];
31      int actPoz;
32      int speed;
33      int health;
34      public Bear(){
35          health = 100;
36          speed = -2;
37          bearPoz = new int [2];
38          bearPoz[0] = -300;
39          bearPoz[1] = 1666;
40          Random rand = new Random();
41          actPoz = rand.nextInt(2);
42          bear = new ImageView();
43          image1 = null;
44          if(actPoz == 1)
45              image1 = new Image(getClass().getClassLoader().getResource("hd/characters/enemy/lvl1/bear/bear_R.gif").toString());
46          else
47              image1 = new Image(getClass().getClassLoader().getResource("hd/characters/enemy/lvl1/bear/bear.gif").toString());
48          bear.setImage(image1);
49          bear.setX(bearPoz[actPoz]);
50          bear.setY(350);
51          knocked = 0;
52          if(actPoz == 0)
53              speed *= -1;
54      }
55      /**
56       * A medve eletet kapjuk meg.
57       * @return a medve elete
58       */
59      public int getHealth(){
60          return health;
61      }
62      /**
63       * Modostijuk a medve eletet.
64       * @param x el modosul
65       */
66      public void setHealth(int x){
67          health = x;
68      }
69      /**
70       * A medvehez tartozo kepet adja vissza.
71       * @return azt az ImageViewet adja vissza amelyet be toltok a scenembe
72       */
73      public ImageView getView(){
74          return bear;
75      }
76      /**
77       * Kep kinyeres.
78       * @return Megkapjuk, hogy aktualisan melyik kep van betoltve az ImageViewunkba 
79       */
80      public Image getImage(){
81          return image1;
82      }
83      /**
84       * Pozicio kinyeres.
85       * @return A medve aktualis pozicioja
86       */
87      public double getPoz(){
88          return bear.getX();
89      }
90      /**
91       * A hatralokes megvalositasa, az alapjan, hogy a konstruktorban mely poziciora tettuk le a medvet.
92       */
93      private void back(){
94          if(actPoz == 1)
95               bear.setX(bear.getX() + 50);
96          else
97              bear.setX(bear.getX() - 50);
98          knocked--;
99      }
100     /**
101      * A medve sebessege.
102      * @return A medve mozgasi sebesseget adja meg
103      */
104     public int getSpeed(){
105         return speed;
106     }
107     /**
108      * Jobbrol utkozes erzekelese.
109      * Megnezzuk, hogy a karakter pozicioja + annak szelessege a medve poziciojahoz kepest hogyan helyezkedik el
110      * X koordinatakat vizsgálok csak.
111      * Egy tartomanyt hasznalok benne, mert ha statikusana cska egy adott X koordinatara vizsgalnek, akkor
112      * a sok mozgasi hatas miatt bekovetkezhet, hogy rosszul tortenik a vizsgalat,
113      * @param flokesz a karakter amihez kepest frissitunk mindent
114      * @return olyan logikai ertek amely visszaadja, hogy hatra kell e lokni a medvet
115      */
116     public boolean toBackR(Character flokesz){
117         return (abs(flokesz.getPoz() + flokesz.getImg().getWidth() - bear.getX()) < 5) 
118                 || (abs(flokesz.getPoz() + flokesz.getImg().getWidth() / 2 - bear.getX()) < 5);
119     }
120     /**
121      * A fent letrehozott hatralokes fuggveny megvalositasa balra.
122      * @param flokesz a karater amihez kepest frissitunk mindent
123      * @return olyan logikai ertek amely megadja, hogy hatra kell-e lokni a medvet
124      */
125     public boolean toBackL(Character flokesz){
126          return (abs(flokesz.getPoz() - (bear.getX() + image1.getWidth())) < 5) 
127                 || (abs(flokesz.getPoz()  - (bear.getX() + image1.getWidth() / 2)) < 5); 
128     }
129     /**
130      * A medve frissitese.
131      * A sebzes megvalositasa.
132      * A hatralokes megvalositasa.
133      * A mozgas megvalositasa.
134      * 
135      * @param flokesz 
136      */
137     public void update(Character flokesz){
138         if(knocked == 0){
139             if(toBackR(flokesz)
140               || toBackL(flokesz)
141                     ){
142             knocked = 8;
143             if(!flokesz.getUtes()){
144                 flokesz.setBacked(8);
145                 flokesz.setHp(-20);
146             }
147             else{
148                 health -= flokesz.getWeapon().getDamage();
149             }
150         }
151         else
152             bear.setX(bear.getX() + speed);
153         }
154         else
155             back();
156         
157     }
158 }
